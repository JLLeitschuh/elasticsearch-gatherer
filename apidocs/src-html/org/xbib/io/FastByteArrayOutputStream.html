002package org.xbib.io;
003
004import org.apache.lucene.util.ArrayUtil;
005import org.elasticsearch.common.bytes.BytesArray;
006import org.elasticsearch.common.bytes.BytesReference;
007
008import java.io.IOException;
009import java.io.OutputStream;
010import java.io.UnsupportedEncodingException;
011import java.util.Arrays;
012
013public class FastByteArrayOutputStream extends OutputStream {
014
015    /**
016     * The buffer where data is stored.
017     */
018    protected byte buf[];
019
020    /**
021     * The number of valid bytes in the buffer.
022     */
023    protected int count;
024
025    /**
026     * Creates a new byte array output stream. The buffer capacity is
027     * initially 1024 bytes, though its size increases if necessary.
028     *
029     * We use 1024 bytes since we mainly use this to build json/smile
030     * content in memory, and rarely does the 32 byte default in ByteArrayOutputStream fits...
031     */
032    public FastByteArrayOutputStream() {
033        this(1024);
034    }
035
036    /**
037     * Creates a new byte array output stream, with a buffer capacity of
038     * the specified size, in bytes.
039     *
040     * @param size the initial size.
041     * @throws IllegalArgumentException if size is negative.
042     */
043    public FastByteArrayOutputStream(int size) {
044        if (size < 0) {
045            throw new IllegalArgumentException("Negative initial size: "
046                    + size);
047        }
048        buf = new byte[size];
049    }
050
051    /**
052     * Writes the specified byte to this byte array output stream.
053     *
054     * @param b the byte to be written.
055     */
056    public void write(int b) {
057        int newcount = count + 1;
058        if (newcount > buf.length) {
059            buf = Arrays.copyOf(buf, ArrayUtil.oversize(newcount, 1));
060        }
061        buf[count] = (byte) b;
062        count = newcount;
063    }
064
065    /**
066     * Writes <code>len</code> bytes from the specified byte array
067     * starting at offset <code>off</code> to this byte array output stream.
068     * <p/>
069     * <b>NO checks for bounds, parameters must be ok!</b>
070     *
071     * @param b   the data.
072     * @param off the start offset in the data.
073     * @param len the number of bytes to write.
074     */
075    public void write(byte b[], int off, int len) {
076        if (len == 0) {
077            return;
078        }
079        int newcount = count + len;
080        if (newcount > buf.length) {
081            buf = Arrays.copyOf(buf, ArrayUtil.oversize(newcount, 1));
082        }
083        System.arraycopy(b, off, buf, count, len);
084        count = newcount;
085    }
086
087    /**
088     * Writes the complete contents of this byte array output stream to
089     * the specified output stream argument, as if by calling the output
090     * stream's write method using <code>out.write(buf, 0, count)</code>.
091     *
092     * @param out the output stream to which to write the data.
093     * @throws java.io.IOException if an I/O error occurs.
094     */
095    public void writeTo(OutputStream out) throws IOException {
096        out.write(buf, 0, count);
097    }
098
099    /**
100     * Resets the <code>count</code> field of this byte array output
101     * stream to zero, so that all currently accumulated output in the
102     * output stream is discarded. The output stream can be used again,
103     * reusing the already allocated buffer space.
104     *
105     * @see java.io.ByteArrayInputStream#count
106     */
107    public void reset() {
108        count = 0;
109    }
110
111    /**
112     * Returns the underlying byte array. Note, use {@link #size()} in order to know
113     * the length of it.
114     */
115    
116    public BytesReference bytes() {
117        return new BytesArray(buf, 0, count);
118    }
119
120    /**
121     * Returns the current size of the buffer.
122     *
123     * @return the value of the <code>count</code> field, which is the number
124     *         of valid bytes in this output stream.
125     * @see java.io.ByteArrayOutputStream#count
126     */
127    public int size() {
128        return count;
129    }
130
131    /**
132     * Seeks back to the given position. Size will become the seeked location.
133     */
134    public void seek(int position) {
135        this.count = position;
136    }
137
138    /**
139     * Converts the buffer's contents into a string decoding bytes using the
140     * platform's default character set. The length of the new <tt>String</tt>
141     * is a function of the character set, and hence may not be equal to the
142     * size of the buffer.
143     * <p/>
144     * <p> This method always replaces malformed-input and unmappable-character
145     * sequences with the default replacement string for the platform's
146     * default character set. The {@linkplain java.nio.charset.CharsetDecoder}
147     * class should be used when more control over the decoding process is
148     * required.
149     *
150     * @return String decoded from the buffer's contents.
151     */
152    public String toString() {
153        return new String(buf, 0, count);
154    }
155
156    /**
157     * Converts the buffer's contents into a string by decoding the bytes using
158     * the specified {@link java.nio.charset.Charset charsetName}. The length of
159     * the new <tt>String</tt> is a function of the charset, and hence may not be
160     * equal to the length of the byte array.
161     * <p/>
162     * <p> This method always replaces malformed-input and unmappable-character
163     * sequences with this charset's default replacement string. The {@link
164     * java.nio.charset.CharsetDecoder} class should be used when more control
165     * over the decoding process is required.
166     *
167     * @param charsetName the name of a supported
168     *                    {@linkplain java.nio.charset.Charset </code>charset<code>}
169     * @return String decoded from the buffer's contents.
170     * @throws java.io.UnsupportedEncodingException
171     *          If the named charset is not supported
172     */
173    public String toString(String charsetName)
174            throws UnsupportedEncodingException {
175        return new String(buf, 0, count, charsetName);
176    }
177
178    /**
179     * Closing a <tt>ByteArrayOutputStream</tt> has no effect. The methods in
180     * this class can be called after the stream has been closed without
181     * generating an <tt>IOException</tt>.
182     * <p/>
183     */
184    public void close() throws IOException {
185    }
186
187
188}
