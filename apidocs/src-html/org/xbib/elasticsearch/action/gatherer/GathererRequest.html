001package org.xbib.elasticsearch.action.gatherer;
002
003import org.elasticsearch.action.support.nodes.NodesOperationRequest;
004import org.elasticsearch.common.io.stream.StreamInput;
005import org.elasticsearch.common.io.stream.StreamOutput;
006
007import java.io.IOException;
008
009public class GathererRequest extends NodesOperationRequest<GathererRequest> {
010
011    private boolean leastBusy;
012
013    private boolean minimumLength;
014
015    public GathererRequest() {
016    }
017
018    public GathererRequest(String... nodeIds) {
019        super(nodeIds);
020    }
021
022    public GathererRequest setLeastBusy(boolean leastBusy) {
023        this.leastBusy = leastBusy;
024        return this;
025    }
026
027    public boolean getLeastBusy() {
028        return leastBusy;
029    }
030
031    public GathererRequest setMinimumLength(boolean minimumLength) {
032        this.minimumLength = minimumLength;
033        return this;
034    }
035
036    public boolean getMinimumLength() {
037        return minimumLength;
038    }
039
040    @Override
041    public void readFrom(StreamInput in) throws IOException {
042        super.readFrom(in);
043    }
044
045    @Override
046    public void writeTo(StreamOutput out) throws IOException {
047        super.writeTo(out);
048    }
049
050}
