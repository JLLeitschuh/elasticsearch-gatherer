002package org.xbib.elasticsearch.action.deploy;
003
004import org.elasticsearch.action.ActionListener;
005import org.elasticsearch.action.support.nodes.NodesOperationRequestBuilder;
006import org.elasticsearch.client.ClusterAdminClient;
007import org.elasticsearch.client.internal.InternalClusterAdminClient;
008
009import java.io.IOException;
010
011public class DeployRequestBuilder extends NodesOperationRequestBuilder<DeployRequest, DeployResponse, DeployRequestBuilder> {
012
013    public DeployRequestBuilder(ClusterAdminClient clusterClient) {
014        super((InternalClusterAdminClient) clusterClient, new DeployRequest());
015    }
016
017    @Override
018    protected void doExecute(ActionListener<DeployResponse> listener) {
019        ((ClusterAdminClient) client).execute(DeployAction.INSTANCE, request, listener);
020    }
021
022    @Override
023    public DeployRequest request() {
024        return this.request;
025    }
026
027    public DeployRequestBuilder setName(String name) throws IOException {
028        request.setName(name);
029        return this;
030    }
031
032    public DeployRequestBuilder setPath(String path) throws IOException {
033        request.setPath(path);
034        return this;
035    }
036}
