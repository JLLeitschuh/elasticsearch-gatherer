002package org.xbib.elasticsearch.gatherer.support;
003
004import java.io.IOException;
005import java.util.Map;
006
007/**
008 * A structured object is composed by an object data source together with
009 * meta data about the object.
010 */
011public interface IndexableObject extends ControlKeys, Comparable<IndexableObject> {
012
013    IndexableObject optype(String optype);
014
015    String optype();
016
017    IndexableObject index(String index);
018
019    String index();
020
021    IndexableObject type(String type);
022
023    String type();
024
025    IndexableObject id(String id);
026
027    String id();
028
029    IndexableObject meta(String key, String value);
030
031    String meta(String key);
032
033    IndexableObject source(Map<String, Object> source);
034
035    Map source();
036
037    String build() throws IOException;
038
039    boolean isEmpty();
040
041}
