002package org.xbib.elasticsearch.gatherer.support;
003
004/**
005 * The names of keys with a special meaning for control.
006 *
007 * Mostly, they map to the Elasticsearch bulk item. The _job column denotes
008 * an ID for the event of a fetch execution.
009 */
010public interface ControlKeys {
011
012    // bulk job name
013    String JOB = "_job";
014
015    // bulk operation type
016    String OPTYPE = "_optype";
017
018    // bulk document
019    String INDEX = "_index";
020
021    String TYPE = "_type";
022
023    String ID = "_id";
024
025    // bulk parameters
026    String VERSION = "_version";
027
028    String ROUTING = "_routing";
029
030    String PERCOLATE = "_percolate";
031
032    String PARENT = "_parent";
033
034    String TIMESTAMP = "_timestamp";
035
036    String TTL = "_ttl";
037
038    // JSON
039    String SOURCE = "_source";
040}
