015/**

017 */
018public class SimpleValueListener<O> implements ValueListener {

020    /**
021     * The current structured object
022     */
023    private IndexableObject current;
024    /**
025     * The object before the current object
026     */
027    private IndexableObject prev;

029    /**
030     * The keys of the values. They are examined for the Elasticsearch index
031     * attributes.
032     */
033    private List<String> keys;

036     * Should not be modified.

038    private char delimiter = '.';

067    /**
068     * Receive values.
069     *
070     * @param values the values
071     * @return this value listener
072     * @throws java.io.IOException
073     */
074    @Override
075    public SimpleValueListener values(List<? extends Object> values) throws IOException {
076        boolean hasSource = false;
077        if (current == null) {
078            current = newObject();
079        }
080        if (prev == null) {
081            prev = newObject();
082        }
083        // execute meta operations on pseudo columns
084        for (int i = 0; i < values.size(); i++) {
085            // v may be null
086            Object o = values.get(i);
087            if (o == null) {
088                continue;
089            }
090            String v = o.toString();
091            // JAVA7: string switch
092            String k = keys.get(i);
093            map(k, v, current);
094            if (IndexableObject.SOURCE.equals(k)) {
095                hasSource = true;
096            }
097        }
098        if (hasSource) {
099            end(current);
100            current = newObject();
101            return this;
102        }
103        // switch to next structured object if current is not equal to previous
104        if (!current.equals(prev) || current.isEmpty()) {
105            prev.source(current.source()); // "steal" source
106            end(prev); // here, the element is being prepared for bulk indexing
107            prev = current;
108            current = newObject();
109        }
110        // create current object from values by sequentially merging the values
111        for (int i = 0; i < keys.size(); i++) {
112            Map map = null;
113            try {
114                map = JsonXContent.jsonXContent.createParser(values.get(i).toString()).mapAndClose();
115            } catch (Exception e) {
116                // ignore
117            }
118            Map m = merge(current.source(), keys.get(i), map != null && map.size() > 0 ? map : values.get(i));
119            current.source(m);
120        }
121        return this;
122    }
123
124    protected void map(String k, String v, IndexableObject current) throws IOException {
125        if (IndexableObject.JOB.equals(k)) {
126            current.meta(IndexableObject.JOB, v);
127        } else if (IndexableObject.OPTYPE.equals(k)) {
128            current.optype(v);
129        } else if (IndexableObject.INDEX.equals(k)) {
130            current.index(v);
131        } else if (IndexableObject.TYPE.equals(k)) {
132            current.type(v);
133        } else if (IndexableObject.ID.equals(k)) {
134            current.id(v);
135        } else if (IndexableObject.VERSION.equals(k)) {
136            current.meta(IndexableObject.VERSION, v);
137        } else if (IndexableObject.ROUTING.equals(k)) {
138            current.meta(IndexableObject.ROUTING, v);
139        } else if (IndexableObject.PERCOLATE.equals(k)) {
140            current.meta(IndexableObject.PERCOLATE, v);
141        } else if (IndexableObject.PARENT.equals(k)) {
142            current.meta(IndexableObject.PARENT, v);
143        } else if (IndexableObject.TIMESTAMP.equals(k)) {
144            current.meta(IndexableObject.TIMESTAMP, v);
145        } else if (IndexableObject.TTL.equals(k)) {
146            current.meta(IndexableObject.TTL, v);
147        } else if (IndexableObject.SOURCE.equals(k)) {
148            current.source(JsonXContent.jsonXContent.createParser(v).mapAndClose());
149        }
150    }

152    /**
153     * End of values.
154     *
155     * @return this value listener
156     * @throws java.io.IOException
157     */
158    public SimpleValueListener end() throws IOException {
159        if (prev != null) {
160            prev.source(current.source());
161            end(prev);
162        }
163        prev = newObject();
164        current = newObject();
165        return this;
166    }

206    /**
207     * Merge key/value pair to a map holding a JSON object. The key consists of
208     * a path pointing to the value position in the JSON object. The key,
209     * representing a path, is divided into head/tail. The recursion terminates
210     * if there is only a head and no tail. In this case, the value is added as
211     * a tuple to the map. If the head key exists, the merge process is
212     * continued by following the path represented by the key. If the path does
213     * not exist, a new map is created. A conflict arises if there is no map at
214     * a head key position. Then, the prefix given in the path is considered
215     * illegal.
216     *
217     * @param map   the map for the JSON object
218     * @param key   the key
219     * @param value the value
220     */
221    protected Map<String, Values<O>> merge(Map map, String key, Object value) {
222        if (ControlKeys.INDEX.equals(key)
223                || ControlKeys.ID.equals(key)
224                || ControlKeys.TYPE.equals(key)
225                || ControlKeys.PARENT.equals(key)) {
226            return map;
227        }
228        int i = key.indexOf(delimiter);
229        String index = null;
230        if (i <= 0) {
231            Matcher matcher = p.matcher(key);
232            boolean isSequence = matcher.matches();
233            String head = key;
234            if (isSequence) {
235                head = matcher.group(1);
236                index = matcher.group(2);
237            }
238            if (index == null || index.isEmpty()) {
239                map.put(head, new Values(map.get(head), value, isSequence));
240            } else {
241                if (!map.containsKey(head)) {
242                    map.put(head, newLinkedList());
243                }
244                Object o = map.get(head);
245                if (o instanceof List) {
246                    List l = (List) o;
247                    int j = l.isEmpty() ? -1 : l.size() - 1;
248                    if (j >= 0) {
249                        Map<String, Values<O>> m = (Map<String, Values<O>>) l.get(j);
250                        if (!m.containsKey(index)) {
251                            l.set(j, merge(m, index, value)); // append
252                        } else {
253                            l.add(merge(new HashMap(), index, value));
254                        }
255                    } else {
256                        l.add(merge(new HashMap(), index, value));
257                    }
258                }
259            }
260        } else {
261            String head = key.substring(0, i);
262            Matcher matcher = p.matcher(head);
263            boolean isSequence = matcher.matches();
264            if (isSequence) {
265                head = matcher.group(1);
266            }
267            String tail = key.substring(i + 1);
268            if (map.containsKey(head)) {
269                Object o = map.get(head);
270                if (o instanceof Map) {
271                    merge((Map<String, Values<O>>) o, tail, value);
272                } else {
273                    throw new IllegalArgumentException("illegal head: " + head);
274                }
275            } else {
276                Map<String, Values<O>> m = new HashMap<String, Values<O>>();
277                map.put(head, m);
278                merge(m, tail, value);
279            }
280        }
281        return map;
282    }
283
284    private final static Pattern p = Pattern.compile("^(.*)\\[(.*?)\\]$");
