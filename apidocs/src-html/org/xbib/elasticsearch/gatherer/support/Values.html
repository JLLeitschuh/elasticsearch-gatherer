002package org.xbib.elasticsearch.gatherer.support;
003
004import org.elasticsearch.common.xcontent.XContentBuilder;
005
006import java.io.IOException;
007
008/**
009 * A ValueSet represents one or many values. Each value is represented at most
010 * once. New values are appended. The ValueSet is the base for building arrays
011 * in JSON.
012 */
013public class Values<O extends Object> {
014
015    private O[] value = (O[]) new Object[0];
016
017    /**
018     * Create a new valueset from an existing valueset adding a value v. If
019     * valueset is null, ignore valueset.
020     *
021     * @param values   the valueset
022     * @param value    the value
023     * @param sequence True if value is to be splitted by commas to multiple values
024     */
025    public Values(Object values, O value, boolean sequence) {
026        if (values instanceof Values) {
027            O[] vals = (O[]) ((Values) values).getValues();
028            if (vals != null) {
029                this.value = vals;
030            }
031        }
032        O[] newValues = sequence && value != null ?
033                (O[]) value.toString().split(",") :
034                (O[]) new Object[]{value};
035        for (O v : newValues) {
036            addValue(v);
037        }
038    }
039
040    /**
041     * Adds the given value
042     *
043     * @param v
044     */
045    public void addValue(O v) {
046        int l = value.length;
047        for (O aValue : value) {
048            if (v != null && v.equals(aValue)) {
049                //value already found, do nothing
050                return;
051            }
052        }
053        if (l == 0) {
054            value = (O[]) new Object[]{v};
055        } else {
056            //never add a null-value to an existing list of values
057            if (v != null) {
058                if (l == 1 && value[0] == null) {
059                    //if there's one existing value and it's null, replace it with the new one
060                    value = (O[]) new Object[]{v};
061                } else {
062                    //otherwise copy the existing value(s) and add the new one
063                    O[] oldValues = value;
064                    value = (O[]) new Object[l + 1];
065                    System.arraycopy(oldValues, 0, value, 0, l);
066                    value[l] = v;
067                }
068            }
069        }
070    }
071
072    /**
073     * Get the values.
074     *
075     * @return the values
076     */
077    public O[] getValues() {
078        return value;
079    }
080
081    public boolean isNull() {
082        return value.length == 0 || (value.length == 1 && value[0] == null);
083    }
084
085    /**
086     * Build ValueSet as XContent
087     *
088     * @param builder the XContentBuilder
089     * @throws java.io.IOException
090     */
091    public void build(XContentBuilder builder) throws IOException {
092        if (value.length > 1) {
093            builder.startArray();
094        }
095        for (O aValue : value) {
096            builder.value(aValue);
097        }
098        if (value.length > 1) {
099            builder.endArray();
100        }
101    }
102
103    @Override
104    public String toString() {
105        StringBuilder sb = new StringBuilder();
106        if (value.length > 1) {
107            sb.append('[');
108        }
109        for (int i = 0; i < value.length; i++) {
110            if (i > 0) {
111                sb.append(',');
112            }
113            sb.append(format(value[i]));
114        }
115        if (value.length > 1) {
116            sb.append(']');
117        }
118        return sb.toString();
119    }
120
121    /**
122     * Format a value for JSON.
123     *
124     * @param o the value
125     * @return the formtted value
126     */
127    private String format(Object o) {
128        if (o == null) {
129            return "null";
130        }
131        if (o instanceof Integer) {
132            return Integer.toString((Integer) o);
133        }
134        if (o instanceof Long) {
135            return Long.toString((Long) o);
136        }
137        if (o instanceof Float) {
138            return Float.toString((Float) o);
139        }
140        if (o instanceof Double) {
141            return Double.toString((Double) o);
142        }
143        if (o instanceof Boolean) {
144            return Boolean.toString((Boolean) o);
145        }
146        // stringify
147        String t = o.toString();
148        t = t.replaceAll("\"", "\\\"");
149        return '"' + t + '"';
150    }
151}
