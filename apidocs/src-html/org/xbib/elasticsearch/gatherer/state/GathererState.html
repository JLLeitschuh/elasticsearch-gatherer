001package org.xbib.elasticsearch.gatherer.state;
002
003import java.io.IOException;
004import java.lang.management.ManagementFactory;
005import java.util.List;
006
007import org.elasticsearch.client.Client;
008import org.elasticsearch.cluster.ClusterChangedEvent;
009import org.elasticsearch.cluster.ClusterService;
010import org.elasticsearch.cluster.ClusterStateListener;
011import org.elasticsearch.common.collect.ImmutableList;
012import org.elasticsearch.common.collect.ImmutableList.Builder;
013import org.elasticsearch.common.component.AbstractComponent;
014import org.elasticsearch.common.inject.Inject;
015import org.elasticsearch.common.settings.ImmutableSettings;
016import org.elasticsearch.common.settings.Settings;
017import org.elasticsearch.common.xcontent.XContentBuilder;
018import org.elasticsearch.common.xcontent.XContentFactory;
019import org.elasticsearch.common.xcontent.XContentParser;
020import org.elasticsearch.monitor.jvm.JvmInfo;
021import org.elasticsearch.transport.Transport;
022import org.elasticsearch.transport.TransportInfo;
023import org.xbib.elasticsearch.gatherer.job.JobEvent;
024
025import static org.elasticsearch.common.xcontent.ToXContent.EMPTY_PARAMS;
026import static org.elasticsearch.common.xcontent.XContentFactory.xContent;
027import static org.elasticsearch.common.xcontent.XContentParser.Token.END_ARRAY;
028import static org.elasticsearch.common.xcontent.XContentType.JSON;
029
030public class GathererState extends AbstractComponent implements ClusterStateListener {
031
032    public static final String PLUGIN_GATHERER_STATE = "plugin.gatherer.state";
033
034    private final Client client;
035
036    private final ClusterService clusterService;
037
038    private final String gathererId;
039
040    @Inject
041    public GathererState(Settings settings, Client client,
042                         ClusterService clusterService,
043                         Transport transport) {
044        super(settings);
045        this.client = client;
046        this.clusterService = clusterService;
047        this.gathererId =  transport.boundAddress() != null ?
048            "gatherer-" + transport.boundAddress().toString() :
049            "gatherer-" + clusterService.localNode().getId();
050    }
051
052    public String getId() {
053        return gathererId;
054    }
055
056    public List<JobEvent> get() throws IOException {
057        return get(PLUGIN_GATHERER_STATE);
058    }
059
060    public void add(JobEvent status) throws IOException {
061        add(PLUGIN_GATHERER_STATE, get(), status);
062    }
063
064    public void remove(JobEvent status) throws IOException {
065        remove(PLUGIN_GATHERER_STATE, get(), status);
066    }
067
068    public void update(JobEvent status) throws IOException {
069        update(PLUGIN_GATHERER_STATE, get(), status);
070    }
071
072    private List<JobEvent> get(String name) throws IOException {
073        return parseSetting(getSetting(name));
074    }
075
076    private void add(String name, List<JobEvent> values, JobEvent status) throws IOException {
077        String value = generateSetting(ImmutableList.<JobEvent>builder()
078                .addAll(values)
079                .add(status)
080                .build());
081        updateSetting(name, value);
082    }
083
084    private void remove(String name, List<JobEvent> values, JobEvent status) throws IOException {
085        Builder<JobEvent> updatedValues = ImmutableList.builder();
086        for (JobEvent value : values) {
087            if (!value.equals(status)) {
088                updatedValues.add(value);
089            }
090        }
091        String value = generateSetting(updatedValues.build());
092        updateSetting(name, value);
093    }
094
095    private void update(String name, List<JobEvent> values, JobEvent status) throws IOException {
096        Builder<JobEvent> updatedValues = ImmutableList.builder();
097        for (JobEvent value : values) {
098            if (value.equals(status)) {
099                updatedValues.add(status);
100            } else {
101                updatedValues.add(value);
102            }
103        }
104        String value = generateSetting(updatedValues.build());
105        updateSetting(name, value);
106    }
107
108    private Settings getSettings() {
109        return clusterService.state().getMetaData().persistentSettings();
110    }
111
112    private String getSetting(String name) {
113        return getSettings().get(name, "[]");
114    }
115
116    private void updateSetting(String name, String value) {
117        client.admin().cluster().prepareUpdateSettings()
118                .setPersistentSettings(ImmutableSettings.builder()
119                        .put(getSettings())
120                        .put(name, value)
121                        .build())
122                .execute()
123                .actionGet();
124    }
125
126    private List<JobEvent> parseSetting(String value) throws IOException {
127        XContentParser parser = xContent(JSON).createParser(value);
128        Builder<JobEvent> builder = ImmutableList.builder();
129        parser.nextToken();
130        while (parser.nextToken() != END_ARRAY) {
131            JobEvent status = new JobEvent();
132            builder.add(status.fromXContent(parser));
133        }
134        return builder.build();
135    }
136
137    private String generateSetting(List<JobEvent> values) throws IOException {
138        XContentBuilder builder = XContentFactory.jsonBuilder();
139        builder.startArray();
140        for (JobEvent value : values) {
141            value.toXContent(builder, EMPTY_PARAMS);
142        }
143        builder.endArray();
144        return builder.string();
145    }
146
147    @Override
148    public void clusterChanged(ClusterChangedEvent event) {
149        logger.info("cluster changed {}", event.state().prettyPrint());
150
151    }
152}
