001package org.xbib.elasticsearch.gatherer;
002
003import org.elasticsearch.ElasticSearchException;
004import org.elasticsearch.common.Strings;
005import org.elasticsearch.common.component.AbstractLifecycleComponent;
006import org.elasticsearch.common.inject.Inject;
007import org.elasticsearch.common.settings.Settings;
008import org.elasticsearch.node.service.NodeService;
009
010import java.lang.management.ManagementFactory;
011import java.lang.management.OperatingSystemMXBean;
012import java.util.concurrent.Executors;
013import java.util.concurrent.ScheduledExecutorService;
014import java.util.concurrent.TimeUnit;
015
016import static org.elasticsearch.common.util.concurrent.EsExecutors.daemonThreadFactory;
017
018/**
019 * The gatherer service manages the gatherer registry.
020 *
021 * Gatherers are announced to the node attributes so they can be seen by node discovery.
022 */
023public class GathererService extends AbstractLifecycleComponent<GathererService> {
024
025    private final GathererRegistry registry;
026
027    private final NodeService nodeService;
028
029    private final OperatingSystemMXBean operatingSystemMXBean;
030
031    @Inject
032    public GathererService(Settings settings,
033                           GathererRegistry registry,
034                           NodeService nodeService) {
035        super(settings);
036        this.registry = registry;
037        this.nodeService = nodeService;
038        this.operatingSystemMXBean = ManagementFactory.getOperatingSystemMXBean();
039    }
040
041    public GathererRegistry getRegistry() {
042        return registry;
043    }
044
045    @Override
046    protected void doStart() throws ElasticSearchException {
047
048        // announce default gatherers
049        announceGatherers();
050
051        // the load updater task
052        ScheduledExecutorService executorService = Executors.newSingleThreadScheduledExecutor(daemonThreadFactory(settings, "gatherer_load_watcher"));
053        executorService.scheduleAtFixedRate(new Thread() {
054            public void run() {
055                double load = operatingSystemMXBean.getSystemLoadAverage();
056                nodeService.putAttribute(GathererConstants.GATHERER_LOAD, Double.toString(load));
057            }
058        }, 1L, 1L, TimeUnit.MINUTES);
059
060        // the queue length updater task
061        executorService.scheduleAtFixedRate(new Thread() {
062            public void run() {
063                int length = 0; // TODO getPendingJobs();
064                nodeService.putAttribute(GathererConstants.GATHERER_LENGTH, Integer.toString(length));
065            }
066        }, 1L, 5L, TimeUnit.SECONDS);
067
068        logger.info("started");
069    }
070
071    @Override
072    protected void doStop() throws ElasticSearchException {
073        nodeService.removeAttribute(GathererConstants.GATHERER_MODULES);
074        nodeService.removeAttribute(GathererConstants.GATHERER_LOAD);
075        nodeService.removeAttribute(GathererConstants.GATHERER_LENGTH);
076    }
077
078    @Override
079    protected void doClose() throws ElasticSearchException {
080    }
081
082    /**
083     * Get all gatherers in the registry and announce them to the node service.
084     */
085    public void announceGatherers() {
086        nodeService.putAttribute(GathererConstants.GATHERER_MODULES,
087                Strings.collectionToCommaDelimitedString(registry.getGatherers().keySet()));
088    }
089
090}
