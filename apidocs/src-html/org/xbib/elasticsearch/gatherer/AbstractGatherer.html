020/**
021 * A base class for gatherers that provides the most used functions.
022 */
023public abstract class AbstractGatherer extends AbstractPlugin implements Gatherer {
024
025    protected final ESLogger logger = ESLoggerFactory.getLogger("gatherer");
026
027    protected Settings settings;
028
029    protected ClusterService clusterService;
030
031    protected Client client;
032
033    protected GathererState state;
034
035    private BulkProcessor bulkProcessor;
036
037    private int bulkActions;
038
039    private ByteSizeValue bulkSize;
040
041    private TimeValue flushInterval;
042
043    private int concurrentRequests;
044
045    public AbstractGatherer() {
046        super();
047    }
048
049    public Gatherer start() throws ElasticSearchException {
050        //this.logger = Loggers.getLogger(getClass(), settings.globalSettings(), riverName);
051        this.bulkActions = settings.getAsInt("bulk_actions", 1000);
052        this.bulkSize = settings.getAsBytesSize("bulk_size", new ByteSizeValue(5, ByteSizeUnit.MB));
053        this.flushInterval = settings.getAsTime("flush_interval", TimeValue.timeValueSeconds(5));
054        this.concurrentRequests = settings.getAsInt("concurrent_requests", 4);
055        bulkProcessor = BulkProcessor.builder(client, new BulkListener())
056                .setBulkActions(bulkActions)
057                .setBulkSize(bulkSize)
058                .setFlushInterval(flushInterval)
059                .setConcurrentRequests(concurrentRequests)
060                .build();
061        return this;
062    }

104
105    class BulkListener implements BulkProcessor.Listener {
106
107        @Override
108        public void beforeBulk(long executionId, BulkRequest request) {
109            if (logger.isTraceEnabled()) {
110                logger.trace("[{}] executing [{}]/[{}]", executionId, request.numberOfActions(), new ByteSizeValue(request.estimatedSizeInBytes()));
111            }
112        }
113
114        @Override
115        public void afterBulk(long executionId, BulkRequest request, BulkResponse response) {
116            if (logger.isTraceEnabled()) {
117                logger.trace("[{}] executed  [{}]/[{}], took [{}]", executionId, request.numberOfActions(), new ByteSizeValue(request.estimatedSizeInBytes()), response.getTook());
118            }
119            if (response.hasFailures()) {
120                logger.warn("[{}] failed to execute bulk request: {}", executionId, response.buildFailureMessage());
121            }
122        }
123
124        @Override
125        public void afterBulk(long executionId, BulkRequest request, Throwable e) {
126            logger.warn("[{}] failed to execute bulk request", e, executionId);
127        }
128    }
