002package org.xbib.cron;
003
004import java.util.Date;
005import java.util.concurrent.CancellationException;
006import java.util.concurrent.RejectedExecutionException;
007import java.util.concurrent.RejectedExecutionHandler;
008import java.util.concurrent.ScheduledThreadPoolExecutor;
009import java.util.concurrent.ThreadFactory;
010import java.util.concurrent.TimeUnit;
011
012/**
013 * Scheduled thread-pool executor implementation that leverages a CronExpression
014 * to calculate future execution times for scheduled tasks.
015 */
016public class CronThreadPoolExecutor extends ScheduledThreadPoolExecutor implements CronExecutorService {
017    /**
018     * Constructs a new CronThreadPoolExecutor.
019     *
020     * @param corePoolSize the pool size
021     */
022    public CronThreadPoolExecutor(int corePoolSize) {
023        super(corePoolSize);
024    }
025
026    /**
027     * Constructs a new CronThreadPoolExecutor.
028     *
029     * @param corePoolSize the pool size
030     */
031    public CronThreadPoolExecutor(int corePoolSize, ThreadFactory threadFactory) {
032        super(corePoolSize, threadFactory);
033    }
034
035    /**
036     * Constructs a new CronThreadPoolExecutor.
037     *
038     * @param corePoolSize the pool size
039     */
040    public CronThreadPoolExecutor(int corePoolSize, RejectedExecutionHandler handler) {
041        super(corePoolSize, handler);
042    }
043
044    /**
045     * Constructs a new CronThreadPoolExecutor.
046     *
047     * @param corePoolSize the pool size
048     * @param handler the handler for rejecting executions
049     */
050    public CronThreadPoolExecutor(int corePoolSize, ThreadFactory threadFactory, RejectedExecutionHandler handler) {
051        super(corePoolSize, threadFactory, handler);
052    }
053
054    @Override
055    public void schedule(final Runnable task, final CronExpression expression) {
056        if (task == null) {
057            throw new NullPointerException();
058        }
059        this.setCorePoolSize(this.getCorePoolSize() + 1);
060        Runnable scheduleTask = new Runnable() {
061            /**
062             * @see Runnable#run()
063             */
064            @Override
065            public void run() {
066                Date now = new Date();
067                Date time = expression.getNextValidTimeAfter(now);
068                try {
069                    while (time != null) {
070                        CronThreadPoolExecutor.this.schedule(task, time.getTime() - now.getTime(), TimeUnit.MILLISECONDS);
071                        while (now.before(time)) {
072                            Thread.sleep(time.getTime() - now.getTime());
073                            now = new Date();
074                        }
075                        time = expression.getNextValidTimeAfter(now);
076                    }
077                } catch (RejectedExecutionException e) {
078                    //
079                } catch (CancellationException e) {
080                    //
081                } catch (InterruptedException e) {
082                    //
083                    Thread.currentThread().interrupt();
084                }
085            }
086        };
087        this.execute(scheduleTask);
088    }
089}
