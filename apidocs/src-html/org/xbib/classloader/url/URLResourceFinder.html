002package org.xbib.classloader.url;
003
004import org.elasticsearch.common.collect.Lists;
005import org.xbib.classloader.ResourceEnumeration;
006import org.xbib.classloader.ResourceFinder;
007import org.xbib.classloader.ResourceHandle;
008import org.xbib.classloader.ResourceLocation;
009import org.xbib.classloader.directory.DirectoryResourceLocation;
010import org.xbib.classloader.jar.JarResourceLocation;
011
012import java.io.File;
013import java.io.FileNotFoundException;
014import java.io.IOException;
015import java.net.MalformedURLException;
016import java.net.URL;
017import java.util.ArrayList;
018import java.util.Arrays;
019import java.util.Collections;
020import java.util.Enumeration;
021import java.util.LinkedHashMap;
022import java.util.LinkedHashSet;
023import java.util.LinkedList;
024import java.util.List;
025import java.util.Map;
026import java.util.StringTokenizer;
027import java.util.jar.Attributes;
028import java.util.jar.Manifest;
029
030public class URLResourceFinder implements ResourceFinder {
031
032    private final Object lock = new Object();
033
034    private final LinkedHashSet<URL> urls = new LinkedHashSet<URL>();
035
036    private final LinkedHashMap<URL, ResourceLocation> classPath = new LinkedHashMap<URL, ResourceLocation>();
037
038    private final LinkedHashSet<File> watchedFiles = new LinkedHashSet<File>();
039
040    private boolean destroyed = false;
041
042    public URLResourceFinder() {
043    }
044
045    public URLResourceFinder(URL[] urls) {
046        addURLs(urls);
047    }
048
049    public void destroy() {
050        synchronized (lock) {
051            if (destroyed) {
052                return;
053            }
054            destroyed = true;
055            urls.clear();
056            for (ResourceLocation resourceLocation : classPath.values()) {
057                resourceLocation.close();
058            }
059            classPath.clear();
060        }
061    }
062
063    public ResourceHandle getResource(String resourceName) {
064        synchronized (lock) {
065            if (destroyed) {
066                return null;
067            }
068            for (Map.Entry<URL, ResourceLocation> entry : getClassPath().entrySet()) {
069                ResourceLocation resourceLocation = entry.getValue();
070                ResourceHandle resourceHandle = resourceLocation.getResourceHandle(resourceName);
071                if (resourceHandle != null && !resourceHandle.isDirectory()) {
072                    return resourceHandle;
073                }
074            }
075        }
076        return null;
077    }
078
079    public URL findResource(String resourceName) {
080        synchronized (lock) {
081            if (destroyed) {
082                return null;
083            }
084            for (Map.Entry<URL, ResourceLocation> entry : getClassPath().entrySet()) {
085                ResourceLocation resourceLocation = entry.getValue();
086                ResourceHandle resourceHandle = resourceLocation.getResourceHandle(resourceName);
087                if (resourceHandle != null) {
088                    return resourceHandle.getUrl();
089                }
090            }
091        }
092        return null;
093    }
094
095    public Enumeration<URL> findResources(String resourceName) {
096        synchronized (lock) {
097            return new ResourceEnumeration(new ArrayList<ResourceLocation>(getClassPath().values()), resourceName);
098        }
099    }
100
101    public void addURL(URL url) {
102        addUrls(Collections.singletonList(url));
103    }
104
105    public URL[] getURLs() {
106        synchronized (lock) {
107            return urls.toArray(new URL[urls.size()]);
108        }
109    }
110
111    /**
112     * Adds an array of urls to the end of this class loader.
113     *
114     * @param urls the URLs to add
115     */
116    public void addURLs(URL[] urls) {
117        addUrls(Arrays.asList(urls));
118    }
119
120    /**
121     * Adds a list of urls to the end of this class loader.
122     *
123     * @param urls the URLs to add
124     */
125    public void addUrls(List<URL> urls) {
126        synchronized (lock) {
127            if (destroyed) {
128                throw new IllegalStateException("UrlResourceFinder has been destroyed");
129            }
130
131            boolean shouldRebuild = this.urls.addAll(urls);
132            if (shouldRebuild) {
133                rebuildClassPath();
134            }
135        }
136    }
137
138    private LinkedHashMap<URL, ResourceLocation> getClassPath() {
139        assert Thread.holdsLock(lock) : "This method can only be called while holding the lock";
140
141        for (File file : watchedFiles) {
142            if (file.canRead()) {
143                rebuildClassPath();
144                break;
145            }
146        }
147
148        return classPath;
149    }
150
151    /**
152     * Rebuilds the entire class path. This class is called when new URLs are
153     * added or one of the watched files becomes readable. This method will not
154     * open jar files again, but will add any new entries not alredy open to the
155     * class path. If any file based url is does not exist, we will watch for
156     * that file to appear.
157     */
158    private void rebuildClassPath() {
159        assert Thread.holdsLock(lock) : "This method can only be called while holding the lock";
160
161        // copy all of the existing locations into a temp map and clear the class path
162        Map<URL, ResourceLocation> existingJarFiles = new LinkedHashMap<URL, ResourceLocation>(classPath);
163        classPath.clear();
164
165        LinkedList<URL> locationStack = new LinkedList<URL>(urls);
166        try {
167            while (!locationStack.isEmpty()) {
168                URL url = locationStack.removeFirst();
169
170                // Skip any duplicate urls in the classpath
171                if (classPath.containsKey(url)) {
172                    continue;
173                }
174
175                // Check is this URL has already been opened
176                ResourceLocation resourceLocation = existingJarFiles.remove(url);
177
178                // If not opened, cache the url and wrap it with a resource location
179                if (resourceLocation == null) {
180                    try {
181                        File file = cacheUrl(url);
182                        resourceLocation = createResourceLocation(url, file);
183                    } catch (FileNotFoundException e) {
184                        // if this is a file URL, the file doesn't exist yet... watch to see if it appears later
185                        if ("file".equals(url.getProtocol())) {
186                            File file = new File(url.getPath());
187                            watchedFiles.add(file);
188                            continue;
189
190                        }
191                    } catch (IOException ignored) {
192                        // can't seem to open the file... this is most likely a bad jar file
193                        // so don't keep a watch out for it because that would require lots of checking
194                        // Dain: We may want to review this decision later
195                        continue;
196                    } catch (UnsupportedOperationException ex) {
197                        // the protocol for the JAR file's URL is not supported.  This can occur when
198                        // the jar file is embedded in an EAR or CAR file.
199                        continue;
200                    }
201                }
202
203                // add the jar to our class path
204                classPath.put(resourceLocation.getCodeSource(), resourceLocation);
205
206                // push the manifest classpath on the stack (make sure to maintain the order)
207                List<URL> manifestClassPath = getManifestClassPath(resourceLocation);
208                locationStack.addAll(0, manifestClassPath);
209            }
210        } catch (Error e) {
211            destroy();
212            throw e;
213        }
214
215        for (ResourceLocation resourceLocation : existingJarFiles.values()) {
216            resourceLocation.close();
217        }
218    }
219
220    protected File cacheUrl(URL url) throws IOException {
221        if (!"file".equals(url.getProtocol())) {
222            // download the jar
223            throw new UnsupportedOperationException("Only local file jars are supported " + url);
224        }
225
226        File file = new File(url.getPath());
227        if (!file.exists()) {
228            throw new FileNotFoundException(file.getAbsolutePath());
229        }
230        if (!file.canRead()) {
231            throw new IOException("File is not readable: " + file.getAbsolutePath());
232        }
233        return file;
234    }
235
236    protected ResourceLocation createResourceLocation(URL codeSource, File cacheFile) throws IOException {
237        if (!cacheFile.exists()) {
238            throw new FileNotFoundException(cacheFile.getAbsolutePath());
239        }
240        if (!cacheFile.canRead()) {
241            throw new IOException("File is not readable: " + cacheFile.getAbsolutePath());
242        }
243
244        ResourceLocation resourceLocation;
245        if (cacheFile.isDirectory()) {
246            // DirectoryResourceLocation will only return "file" URLs within this directory
247            // do not user the DirectoryResourceLocation for non file based urls
248            resourceLocation = new DirectoryResourceLocation(cacheFile);
249        } else {
250            resourceLocation = new JarResourceLocation(codeSource, cacheFile);
251        }
252        return resourceLocation;
253    }
254
255    private List<URL> getManifestClassPath(ResourceLocation resourceLocation) {
256        try {
257            // get the manifest, if possible
258            Manifest manifest = resourceLocation.getManifest();
259            if (manifest == null) {
260                // some locations don't have a manifest
261                return Lists.newLinkedList();
262            }
263
264            // get the class-path attribute, if possible
265            String manifestClassPath = manifest.getMainAttributes().getValue(Attributes.Name.CLASS_PATH);
266            if (manifestClassPath == null) {
267                return Lists.newLinkedList();
268            }
269
270            // build the urls...
271            // the class-path attribute is space delimited
272            URL codeSource = resourceLocation.getCodeSource();
273            LinkedList<URL> classPathUrls = new LinkedList<URL>();
274            for (StringTokenizer tokenizer = new StringTokenizer(manifestClassPath, " "); tokenizer.hasMoreTokens(); ) {
275                String entry = tokenizer.nextToken();
276                try {
277                    // the class path entry is relative to the resource location code source
278                    URL entryUrl = new URL(codeSource, entry);
279                    classPathUrls.addLast(entryUrl);
280                } catch (MalformedURLException ignored) {
281                    // most likely a poorly named entry
282                }
283            }
284            return classPathUrls;
285        } catch (IOException ignored) {
286            // error opening the manifest
287            return Lists.newLinkedList();
288        }
289    }
