002package org.xbib.classloader.directory;
003
004import org.xbib.classloader.ResourceHandle;
005import org.xbib.classloader.url.AbstractURLResourceLocation;
006
007import java.io.File;
008import java.io.FileInputStream;
009import java.io.IOException;
010import java.net.MalformedURLException;
011import java.util.jar.Manifest;
012
013public class DirectoryResourceLocation extends AbstractURLResourceLocation {
014
015    private final File baseDir;
016
017    private boolean manifestLoaded = false;
018
019    private Manifest manifest;
020
021    public DirectoryResourceLocation(File baseDir) throws MalformedURLException {
022        super(baseDir.toURI().toURL());
023        this.baseDir = baseDir;
024    }
025
026    @Override
027    public ResourceHandle getResourceHandle(String resourceName) {
028        File file = new File(baseDir, resourceName);
029        if (!file.exists() || !isLocal(file)) {
030            return null;
031        }
032        try {
033            return new DirectoryResourceHandle(resourceName, file, baseDir, getManifestSafe());
034        } catch (MalformedURLException e) {
035            return null;
036        }
037    }
038
039    private boolean isLocal(File file) {
040        try {
041            String base = baseDir.getCanonicalPath();
042            String relative = file.getCanonicalPath();
043            return (relative.startsWith(base));
044        } catch (IOException e) {
045            return false;
046        }
047    }
048
049    @Override
050    public Manifest getManifest() throws IOException {
051        if (!manifestLoaded) {
052            File manifestFile = new File(baseDir, "META-INF/MANIFEST.MF");
053            if (manifestFile.isFile() && manifestFile.canRead()) {
054                FileInputStream in = null;
055                try {
056                    in = new FileInputStream(manifestFile);
057                    manifest = new Manifest(in);
058                } finally {
059                    if (in != null) {
060                        try {
061                            in.close();
062                        } catch (Exception ignored) {
063                        }
064                    }
065                }
066            }
067            manifestLoaded = true;
068        }
069        return manifest;
070    }
071
072    private Manifest getManifestSafe() {
073        Manifest m = null;
074        try {
075            m = getManifest();
076        } catch (IOException e) {
077            // ignore
078        }
079        return m;
080    }
081}
