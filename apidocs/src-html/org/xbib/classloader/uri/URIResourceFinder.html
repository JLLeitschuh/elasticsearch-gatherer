002package org.xbib.classloader.uri;
003
004import org.elasticsearch.common.collect.Lists;
005import org.xbib.classloader.ResourceEnumeration;
006import org.xbib.classloader.ResourceFinder;
007import org.xbib.classloader.ResourceHandle;
008import org.xbib.classloader.ResourceLocation;
009import org.xbib.classloader.directory.DirectoryResourceLocation;
010import org.xbib.classloader.jar.JarResourceLocation;
011
012import java.io.File;
013import java.io.FileNotFoundException;
014import java.io.IOException;
015import java.net.MalformedURLException;
016import java.net.URI;
017import java.net.URISyntaxException;
018import java.net.URL;
019import java.util.ArrayList;
020import java.util.Arrays;
021import java.util.Collections;
022import java.util.Enumeration;
023import java.util.LinkedHashMap;
024import java.util.LinkedHashSet;
025import java.util.LinkedList;
026import java.util.List;
027import java.util.Map;
028import java.util.StringTokenizer;
029import java.util.jar.Attributes;
030import java.util.jar.Manifest;
031
032public class URIResourceFinder implements ResourceFinder {
033
034    private final Object lock = new Object();
035
036    private final LinkedHashSet<URI> uris = new LinkedHashSet<URI>();
037
038    private final LinkedHashMap<URI, ResourceLocation> classPath = new LinkedHashMap<URI, ResourceLocation>();
039
040    private final LinkedHashSet<File> watchedFiles = new LinkedHashSet<File>();
041
042    private boolean destroyed = false;
043
044    public URIResourceFinder() {
045    }
046
047    public URIResourceFinder(URI[] uris) {
048        addURIs(uris);
049    }
050
051    public void destroy() {
052        synchronized (lock) {
053            if (destroyed) {
054                return;
055            }
056            destroyed = true;
057            uris.clear();
058            for (ResourceLocation resourceLocation : classPath.values()) {
059                resourceLocation.close();
060            }
061            classPath.clear();
062        }
063    }
064
065    public ResourceHandle getResource(String resourceName) {
066        synchronized (lock) {
067            if (destroyed) {
068                return null;
069            }
070            for (Map.Entry<URI, ResourceLocation> entry : getClassPath().entrySet()) {
071                ResourceLocation resourceLocation = entry.getValue();
072                ResourceHandle resourceHandle = resourceLocation.getResourceHandle(resourceName);
073                if (resourceHandle != null && !resourceHandle.isDirectory()) {
074                    return resourceHandle;
075                }
076            }
077        }
078        return null;
079    }
080
081    public URL findResource(String resourceName) {
082        synchronized (lock) {
083            if (destroyed) {
084                return null;
085            }
086            for (Map.Entry<URI, ResourceLocation> entry : getClassPath().entrySet()) {
087                ResourceLocation resourceLocation = entry.getValue();
088                ResourceHandle resourceHandle = resourceLocation.getResourceHandle(resourceName);
089                if (resourceHandle != null) {
090                    return resourceHandle.getUrl();
091                }
092            }
093        }
094        return null;
095    }
096
097    public Enumeration<URL> findResources(String resourceName) {
098        synchronized (lock) {
099            return new ResourceEnumeration(new ArrayList<ResourceLocation>(getClassPath().values()), resourceName);
100        }
101    }
102
103    public void addURI(URI uri) {
104        addUris(Collections.singletonList(uri));
105    }
106
107    public URI[] getURIs() {
108        synchronized (lock) {
109            return uris.toArray(new URI[uris.size()]);
110        }
111    }
112
113    /**
114     * Adds an array of uris to the end of this class loader.
115     *
116     * @param uris the URLs to add
117     */
118    protected void addURIs(URI[] uris) {
119        addUris(Arrays.asList(uris));
120    }
121
122    /**
123     * Adds a list of uris to the end of this class loader.
124     *
125     * @param uris the URLs to add
126     */
127    protected void addUris(List<URI> uris) {
128        synchronized (lock) {
129            if (destroyed) {
130                throw new IllegalStateException("UriResourceFinder has been destroyed");
131            }
132            boolean shouldRebuild = this.uris.addAll(uris);
133            if (shouldRebuild) {
134                rebuildClassPath();
135            }
136        }
137    }
138
139    private LinkedHashMap<URI, ResourceLocation> getClassPath() {
140        assert Thread.holdsLock(lock) : "This method can only be called while holding the lock";
141        for (File file : watchedFiles) {
142            if (file.canRead()) {
143                rebuildClassPath();
144                break;
145            }
146        }
147
148        return classPath;
149    }
150
151    /**
152     * Rebuilds the entire class path. This class is called when new URIs are
153     * added or one of the watched files becomes readable. This method will not
154     * open jar files again, but will add any new entries not alredy open to the
155     * class path. If any file based uri is does not exist, we will watch for
156     * that file to appear.
157     */
158    private void rebuildClassPath() {
159        assert Thread.holdsLock(lock) : "This method can only be called while holding the lock";
160        // copy all of the existing locations into a temp map and clear the class path
161        Map<URI, ResourceLocation> existingJarFiles = new LinkedHashMap<URI, ResourceLocation>(classPath);
162        classPath.clear();
163        LinkedList<URI> locationStack = new LinkedList<URI>(uris);
164        try {
165            while (!locationStack.isEmpty()) {
166                URI uri = locationStack.removeFirst();
167                // Skip any duplicate uris in the classpath
168                if (classPath.containsKey(uri)) {
169                    continue;
170                }
171                // Check is this URL has already been opened
172                ResourceLocation resourceLocation = existingJarFiles.remove(uri);
173                // If not opened, cache the uri and wrap it with a resource location
174                if (resourceLocation == null) {
175                    try {
176                        resourceLocation = createResourceLocation(uri.toURL(), cacheUri(uri));
177                    } catch (FileNotFoundException e) {
178                        // if this is a file URL, the file doesn't exist yet... watch to see if it appears later
179                        if ("file".equals(uri.getScheme())) {
180                            File file = new File(uri.getPath());
181                            watchedFiles.add(file);
182                            continue;
183
184                        }
185                    } catch (IOException ignored) {
186                        // can't seem to open the file... this is most likely a bad jar file
187                        // so don't keep a watch out for it because that would require lots of checking
188                        // Dain: We may want to review this decision later
189                        continue;
190                    } catch (UnsupportedOperationException ex) {
191                        // the protocol for the JAR file's URL is not supported.  This can occur when
192                        // the jar file is embedded in an EAR or CAR file.
193                        continue;
194                    }
195                    try {
196                        // add the jar to our class path
197                        if (resourceLocation != null && resourceLocation.getCodeSource() != null) {
198                            classPath.put(resourceLocation.getCodeSource().toURI(), resourceLocation);
199                        }
200                    } catch (URISyntaxException ex) {
201                        // ignore
202                    }
203                }
204                // push the manifest classpath on the stack (make sure to maintain the order)
205                List<URI> manifestClassPath = getManifestClassPath(resourceLocation);
206                locationStack.addAll(0, manifestClassPath);
207            }
208        } catch (Error e) {
209            destroy();
210            throw e;
211        }
212        for (ResourceLocation resourceLocation : existingJarFiles.values()) {
213            resourceLocation.close();
214        }
215    }
216
217    protected File cacheUri(URI uri) throws IOException {
218        if (!"file".equals(uri.getScheme())) {
219            // download the jar
220            throw new UnsupportedOperationException("Only local file jars are supported " + uri);
221        }
222        File file = new File(uri.getPath());
223        if (!file.exists()) {
224            throw new FileNotFoundException(file.getAbsolutePath());
225        }
226        if (!file.canRead()) {
227            throw new IOException("File is not readable: " + file.getAbsolutePath());
228        }
229        return file;
230    }
231
232    protected ResourceLocation createResourceLocation(URL codeSource, File cacheFile) throws IOException {
233        if (!cacheFile.exists()) {
234            throw new FileNotFoundException(cacheFile.getAbsolutePath());
235        }
236        if (!cacheFile.canRead()) {
237            throw new IOException("File is not readable: " + cacheFile.getAbsolutePath());
238        }
239        return cacheFile.isDirectory() ?
240            // DirectoryResourceLocation will only return "file" URLs within this directory
241            // do not use the DirectoryResourceLocation for non file based uris
242            new DirectoryResourceLocation(cacheFile) :
243            new JarResourceLocation(codeSource, cacheFile);
244    }
245
246    private List<URI> getManifestClassPath(ResourceLocation resourceLocation) {
247        try {
248            // get the manifest, if possible
249            Manifest manifest = resourceLocation.getManifest();
250            if (manifest == null) {
251                // some locations don't have a manifest
252                return Lists.newLinkedList();
253            }
254
255            // get the class-path attribute, if possible
256            String manifestClassPath = manifest.getMainAttributes().getValue(Attributes.Name.CLASS_PATH);
257            if (manifestClassPath == null) {
258                return Lists.newLinkedList();
259            }
260
261            // build the uris...
262            // the class-path attribute is space delimited
263            URL codeSource = resourceLocation.getCodeSource();
264            LinkedList<URI> classPathUrls = new LinkedList<URI>();
265            for (StringTokenizer tokenizer = new StringTokenizer(manifestClassPath, " "); tokenizer.hasMoreTokens(); ) {
266                String entry = tokenizer.nextToken();
267                try {
268                    // the class path entry is relative to the resource location code source
269                    URL entryUrl = new URL(codeSource, entry);
270                    classPathUrls.addLast(entryUrl.toURI());
271                } catch (MalformedURLException ignored) {
272                    // most likely a poorly named entry
273                } catch (URISyntaxException ignored) {
274                    // most likely a poorly named entry
275                }
276            }
277            return classPathUrls;
278        } catch (IOException ignored) {
279            // error opening the manifest
280            return Lists.newLinkedList();
281        }
282    }
