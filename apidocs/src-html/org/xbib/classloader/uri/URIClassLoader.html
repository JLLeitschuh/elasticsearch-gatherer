002package org.xbib.classloader.uri;

004import org.xbib.classloader.ResourceHandle;

006import java.io.File;
007import java.io.IOException;
008import java.net.URI;
009import java.net.URISyntaxException;
010import java.net.URL;
011import java.net.URLClassLoader;
012import java.security.AccessControlContext;
013import java.security.AccessController;
014import java.security.PrivilegedActionException;
015import java.security.cert.Certificate;
016import java.security.CodeSource;
017import java.security.PrivilegedAction;
018import java.security.PrivilegedExceptionAction;
019import java.util.Enumeration;
020import java.util.jar.Attributes;
021import java.util.jar.Attributes.Name;
022import java.util.jar.Manifest;

024/**
025 * Equivalent of java.net.URLClassloader but without bugs related to ill-formed
026 * URLs and with customizable JAR caching policy. The standard URLClassLoader
027 * accepts URLs containing spaces and other characters which are forbidden in
028 * the URI syntax, according to the RFC 2396. As a workaround to this problem,
029 * Java escapes and un-escapes URLs in various arbitrary places; however, this
030 * is inconsistent and leads to numerous problems with URLs referring to local
031 * files with spaces in the path. SUN acknowledges the problem, but refuses to
032 * modify the behavior for compatibility reasons; see Java Bug Parade 4273532,
033 * 4466485.
034 * <p/>
035 * Additionally, the JAR caching policy used by URLClassLoader is system-wide
036 * and inflexible: once downloaded JAR files are never re-downloaded, even if
037 * one creates a fresh instance of the class loader that happens to have the
038 * same URL in its search path. In fact, that policy is a security
039 * vulnerability: it is possible to crash any URL class loader, thus affecting
040 * potentially separate part of the system, by creating URL connection to one of
041 * the URLs of that class loader search path and closing the associated JAR
042 * file. See Java Bug Parade 4405789, 4388666, 4639900.
043 * <p/>
044 * This class avoids these problems by 1) using URIs instead of URLs for the
045 * search path (thus enforcing strict syntax conformance and defining precise
046 * escaping semantics), and 2) using custom URLStreamHandler which ensures
047 * per-classloader JAR caching policy.
048 */
049public final class URIClassLoader extends URLClassLoader {
050
051    private final URIResourceFinder finder = new URIResourceFinder();
052
053    private final AccessControlContext acc;
054
055    /**
056     * Creates URIClassLoader
057     */
058    public URIClassLoader() {
059        this(null);
060    }
061
062    /**
063     * Creates URIClassLoader with the specified parent class loader.
064     *
065     * @param parent the parent class loader.
066     */
067    public URIClassLoader(ClassLoader parent) {
068        super(new URL[0], parent);
069        this.acc = AccessController.getContext();
070    }
071
072    /**
073     * Add specified URI at the end of the search path.
074     *
075     * @param uri the URI to add
076     */
077    public URIClassLoader addURI(URI uri) {
078        finder.addURI(uri);
079        return this;
080    }
081
082    public URI[] getURIs() {
083        return finder.getURIs();
084    }
085
086    /**
087     * Add specified URL at the end of the search path.
088     *
089     * @param url the URL to add
090     * @deprecated use addURI
091     */
092    @Override
093    @Deprecated
094    protected void addURL(URL url) {
095        try {
096            finder.addURI(url.toURI());
097        } catch (URISyntaxException e) {
098            // ignore
099        }
100    }
101
102    /**
103     * Finds and loads the class with the specified name.
104     *
105     * @param name the name of the class
106     * @return the resulting class
107     * @throws ClassNotFoundException if the class could not be found
108     */
109    @Override
110    protected Class findClass(final String name) throws ClassNotFoundException {
111        try {
112            return (Class) AccessController.doPrivileged(new PrivilegedExceptionAction() {
113                @Override
114                public Object run() throws ClassNotFoundException {
115                    String path = name.replace('.', '/').concat(".class");
116                    ResourceHandle h = finder.getResource(path);
117                    if (h != null) {
118                        try {
119                            return defineClass(name, h);
120                        } catch (IOException e) {
121                            throw new ClassNotFoundException(name, e);
122                        }
123                    } else {
124                        throw new ClassNotFoundException(name);
125                    }
126                }
127            }, acc);
128        } catch (PrivilegedActionException pae) {
129            throw (ClassNotFoundException) pae.getException();
130        }
131    }
132
133    /**
134     * Finds the resource with the specified name.
135     *
136     * @param name the name of the resource
137     * @return a <code>URL</code> for the resource, or <code>null</code> if the
138     * resource could not be found.
139     */
140    @Override
141    public URL findResource(final String name) {
142        return (URL) AccessController.doPrivileged(new PrivilegedAction() {
143            public Object run() {
144                return finder.findResource(name);
145            }
146        }, acc);
147    }
148
149    /**
150     * Returns an Enumeration of URLs representing all of the resources having
151     * the specified name.
152     *
153     * @param name the resource name
154     * @return an <code>Enumeration</code> of <code>URL</code>s
155     * @throws java.io.IOException if an I/O exception occurs
156     */
157    @Override
158    public Enumeration<URL> findResources(final String name) throws IOException {
159        return (Enumeration<URL>) AccessController.doPrivileged(new PrivilegedAction() {
160            public Object run() {
161                return finder.findResources(name);
162            }
163        }, acc);
164    }
165
166    /**
167     * Returns the absolute path name of a native library. The VM invokes this
168     * method to locate the native libraries that belong to classes loaded with
169     * this class loader. If this method returns
170     * <code>null</code>, the VM searches the library along the path specified
171     * as the
172     * <code>java.library.path</code> property. This method invoke
173     * {@link #getLibraryHandle} method to find handle of this library. If the
174     * handle is found and its URL protocol is "file", the system-dependent
175     * absolute library file path is returned. Otherwise this method returns
176     * null. <p>
177     * <p/>
178     * Subclasses can override this method to provide specific approaches in
179     * library searching.
180     *
181     * @param libname the library name
182     * @return the absolute path of the native library
183     * @see System#loadLibrary(String)
184     * @see System#mapLibraryName(String)
185     */
186    @Override
187    protected String findLibrary(String libname) {
188        ResourceHandle md = getLibraryHandle(libname);
189        if (md == null) {
190            return null;
191        }
192        URL url = md.getUrl();
193        if (!"file".equals(url.getProtocol())) {
194            return null;
195        }
196        return new File(URI.create(url.toString())).getPath();
197    }
198
199    /**
200     * Finds the ResourceHandle object for the resource with the specified name.
201     *
202     * @param name the name of the resource
203     * @return the ResourceHandle of the resource
204     */
205    private ResourceHandle getResourceHandle(final String name) {
206        return (ResourceHandle) AccessController.doPrivileged(new PrivilegedAction() {
207            public Object run() {
208                return finder.getResource(name);
209            }
210        }, acc);
211    }
212
213    /**
214     * Finds the ResourceHandle object for the native library with the specified
215     * name. The library name must be '/'-separated path. The last part of this
216     * path is substituted by its system-dependent mapping (using
217     * {@link System#mapLibraryName(String)} method). Next, the
218     * <code>ResourceFinder</code> is used to look for the library as it was
219     * ordinary resource. <p>
220     * <p/>
221     * Subclasses can override this method to provide specific approaches in
222     * library searching.
223     *
224     * @param name the name of the library
225     * @return the ResourceHandle of the library
226     */
227    private ResourceHandle getLibraryHandle(final String name) {
228        int idx = name.lastIndexOf('/');
229        String path;
230        String simplename;
231        if (idx == -1) {
232            path = "";
233            simplename = name;
234        } else if (idx == name.length() - 1) { // name.endsWith('/')
235            throw new IllegalArgumentException(name);
236        } else {
237            path = name.substring(0, idx + 1); // including '/'
238            simplename = name.substring(idx + 1);
239        }
240        return getResourceHandle(path + System.mapLibraryName(simplename));
241    }
242
243    /**
244     * Returns true if the specified package name is sealed according to the
245     * given manifest.
246     */
247    private boolean isSealed(String name, Manifest man) {
248        String path = name.replace('.', '/').concat("/");
249        Attributes attr = man.getAttributes(path);
250        String sealed = null;
251        if (attr != null) {
252            sealed = attr.getValue(Name.SEALED);
253        }
254        if (sealed == null) {
255            if ((attr = man.getMainAttributes()) != null) {
256                sealed = attr.getValue(Name.SEALED);
257            }
258        }
259        return "true".equalsIgnoreCase(sealed);
260    }
261
262    private Class defineClass(String name, ResourceHandle h) throws IOException {
263        int i = name.lastIndexOf('.');
264        URL url = h.getCodeSourceUrl();
265        if (i != -1) { // check package
266            String pkgname = name.substring(0, i);
267            // check if package already loaded
268            Package pkg = getPackage(pkgname);
269            Manifest man = h.getManifest();
270            if (pkg != null) {
271                // package found, so check package sealing
272                boolean ok;
273                if (pkg.isSealed()) {
274                    // verify that code source URLs are the same
275                    ok = pkg.isSealed(url);
276                } else {
277                    // make sure we are not attempting to seal the package
278                    // at this code source URL
279                    ok = (man == null) || !isSealed(pkgname, man);
280                }
281                if (!ok) {
282                    throw new SecurityException("sealing violation: " + name);
283                }
284            } else { // package not yet defined
285                if (man != null) {
286                    definePackage(pkgname, man, url);
287                } else {
288                    definePackage(pkgname, null, null, null, null, null, null, null);
289                }
290            }
291        }
292        // now read the class bytes and define the class
293        byte[] b = h.getBytes();
294        Certificate[] certs = h.getCertificates();
295        CodeSource cs = new CodeSource(url, certs);
296        return defineClass(name, b, 0, b.length, cs);
297    }
