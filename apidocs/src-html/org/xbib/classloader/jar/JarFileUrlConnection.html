002package org.xbib.classloader.jar;
003
004import java.io.File;
005import java.io.IOException;
006import java.io.InputStream;
007import java.net.JarURLConnection;
008import java.net.MalformedURLException;
009import java.net.URL;
010import java.net.URLConnection;
011import java.net.URLStreamHandler;
012import java.security.Permission;
013import java.security.cert.Certificate;
014import java.util.jar.Attributes;
015import java.util.jar.JarEntry;
016import java.util.jar.JarFile;
017import java.util.jar.Manifest;
018
019public class JarFileUrlConnection extends JarURLConnection {
020
021    public static final URL DUMMY_JAR_URL;
022
023    static {
024        try {
025            DUMMY_JAR_URL = new URL("jar", "", -1, "file:dummy!/", new URLStreamHandler() {
026                protected URLConnection openConnection(URL u) {
027                    throw new UnsupportedOperationException();
028                }
029            });
030        } catch (Exception e) {
031            throw new ExceptionInInitializerError(e);
032        }
033    }
034
035    private final URL url;
036
037    private final JarFile jarFile;
038
039    private final JarEntry jarEntry;
040
041    private final URL jarFileUrl;
042
043    public JarFileUrlConnection(URL url, JarFile jarFile, JarEntry jarEntry) throws MalformedURLException {
044        super(DUMMY_JAR_URL);
045        if (url == null) {
046            throw new NullPointerException("url is null");
047        }
048        if (jarFile == null) {
049            throw new NullPointerException("jarFile is null");
050        }
051        if (jarEntry == null) {
052            throw new NullPointerException("jarEntry is null");
053        }
054        this.url = url;
055        this.jarFile = jarFile;
056        this.jarEntry = jarEntry;
057        jarFileUrl = new File(jarFile.getName()).toURI().toURL();
058    }
059
060    @Override
061    public JarFile getJarFile() throws IOException {
062        if (getUseCaches()) {
063            return jarFile;
064        } else {
065            return new JarFile(jarFile.getName());
066        }
067    }
068
069    @Override
070    public synchronized void connect() {
071    }
072
073    @Override
074    public URL getJarFileURL() {
075        return jarFileUrl;
076    }
077
078    @Override
079    public String getEntryName() {
080        return getJarEntry().getName();
081    }
082
083    @Override
084    public Manifest getManifest() throws IOException {
085        return jarFile.getManifest();
086    }
087
088    @Override
089    public JarEntry getJarEntry() {
090        if (getUseCaches()) {
091            return jarEntry;
092        } else {
093            //return (JarEntry) jarEntry.clone();
094            // There is a clone method, but the below way might be safer.
095            return jarFile.getJarEntry(jarEntry.getName());
096        }
097    }
098
099    @Override
100    public Attributes getAttributes() throws IOException {
101        return getJarEntry().getAttributes();
102    }
103
104    @Override
105    public Attributes getMainAttributes() throws IOException {
106        return getManifest().getMainAttributes();
107    }
108
109    @Override
110    public Certificate[] getCertificates() throws IOException {
111        return getJarEntry().getCertificates();
112    }
113
114    @Override
115    public URL getURL() {
116        return url;
117    }
118
119    @Override
120    public int getContentLength() {
121        long size = getJarEntry().getSize();
122        if (size > Integer.MAX_VALUE) {
123            return -1;
124        }
125        return (int) size;
126    }
127
128    @Override
129    public long getLastModified() {
130        return getJarEntry().getTime();
131    }
132
133    @Override
134    public synchronized InputStream getInputStream() throws IOException {
135        return jarFile.getInputStream(jarEntry);
136    }
137
138    @Override
139    public Permission getPermission() throws IOException {
140        URL jarFileUrl = new File(jarFile.getName()).toURI().toURL();
141        return jarFileUrl.openConnection().getPermission();
142    }
143
144    @Override
145    public String toString() {
146        return JarFileUrlConnection.class.getName() + ":" + url;
147    }
148}
