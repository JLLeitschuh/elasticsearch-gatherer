002package org.xbib.classloader.jar;
003
004import org.xbib.classloader.AbstractResourceHandle;
005
006import java.io.IOException;
007import java.io.InputStream;
008import java.net.MalformedURLException;
009import java.net.URL;
010import java.security.cert.Certificate;
011import java.util.jar.Attributes;
012import java.util.jar.JarEntry;
013import java.util.jar.JarFile;
014import java.util.jar.Manifest;
015
016public class JarResourceHandle extends AbstractResourceHandle {
017
018    private final JarFile jarFile;
019
020    private final JarEntry jarEntry;
021
022    private final URL url;
023
024    private final URL codeSource;
025
026    public JarResourceHandle(JarFile jarFile, JarEntry jarEntry, URL codeSource) throws MalformedURLException {
027        this.jarFile = jarFile;
028        this.jarEntry = jarEntry;
029        this.url = JarFileUrlStreamHandler.createURL(jarFile, jarEntry, codeSource);
030        this.codeSource = codeSource;
031    }
032
033    public String getName() {
034        return jarEntry.getName();
035    }
036
037    public URL getUrl() {
038        return url;
039    }
040
041    public URL getCodeSourceUrl() {
042        return codeSource;
043    }
044
045    public boolean isDirectory() {
046        return jarEntry.isDirectory();
047    }
048
049    public InputStream getInputStream() throws IOException {
050        return jarFile.getInputStream(jarEntry);
051    }
052
053    public int getContentLength() {
054        return (int) jarEntry.getSize();
055    }
056
057    @Override
058    public Manifest getManifest() throws IOException {
059        return jarFile.getManifest();
060    }
061
062    @Override
063    public Attributes getAttributes() throws IOException {
064        return jarEntry.getAttributes();
065    }
066
067    @Override
068    public Certificate[] getCertificates() {
069        return jarEntry.getCertificates();
070    }
071}
